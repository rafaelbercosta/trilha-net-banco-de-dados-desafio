12-	SELECT 
	    f.Nome        AS Filme,
	    a.PrimeiroNome,
	    a.UltimoNome,
	    ef.Papel
	FROM 
	    Filmes f
	INNER JOIN 
	    ElencoFilme ef ON f.Id = ef.IdFilme
	INNER JOIN 
	    Atores a ON ef.IdAtor = a.Id
	ORDER BY 
	    f.Nome, a.PrimeiroNome;
